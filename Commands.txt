# ###########################
# Add a new remote to GitHub
# ###########################
git remote add origin https://github.com/geridantech/GIT.git

# Set a new remote
git remote -v

# Reset latest commit
git reset HEAD~

# Verify new remote
origin  https://github.com/geridantech/GIT.git (fetch)
origin  https://github.com/geridantech/GIT.git (push)

# Rename a Remote Branch
git remote rename <old> <new>

# Create new branch on local, switch to branch
$ git checkout -b [name_of_your_new_branch]

# Check current file status
$ git status

# Add Username / Email to git config settings
$git config --global user.(name/email)

# Remove large files from history
https://help.github.com/articles/removing-files-from-a-repository-s-history/

# Specify remote, specify remote name [origin] and link to [repository]
git set -u origin repository

# Remove remote named [origin]
git remote remove origin
